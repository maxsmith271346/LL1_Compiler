// Mutual recursion with conditionals

main

int n;

function foo ( int x ) : bool
{
    if (x == 0) then
        return true;
    else
        return call bar(x - 1);
    fi;
};

function bar ( int x ) : bool
{
    if (x == 0) then
        return false;
    else
        return call foo(x - 1);
    fi;
};

{
    let n = 8;
    if (n >= 0) then
        call printBool(call foo(n));
    else
        call printInt(n);
    fi;
} .
