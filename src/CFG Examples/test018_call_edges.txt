main

function foo() : void
{
	int a, b, c, d, e, f, g, h;

	let c = 1;
//	let c = a + b;
	let d = c;
	let e = d * d;
	let f = a + c;
	let g = e;
	let a = g + d;
	
	if (a < c) then
		let h = g + 1;
	else
		let f = d + g;
		if (f > a) then
			let c = 2;
		fi;
	fi;
	
	let b = g * a;
	
	call printInt(a);
	call printInt(b);
	call printInt(c);
	call printInt(d);
	call printInt(e);
	call printInt(f);
	call printInt(g);
	call printInt(h);
};

{
	call foo();
}
