// Repeat Statement
main

int i, n;
bool flag;

{

    let n = call readInt();
    let i = 1;
    let flag = false;

    if (n > 1) then
        repeat
            let i+=1;
            call printInt(i);
            if (n % i == 0) then
                let flag = true;
            fi;
        until (i >= (n - 1));
        call printInt(n);
    fi;

    //call printInt(i);
    call printBool((flag));
}.
